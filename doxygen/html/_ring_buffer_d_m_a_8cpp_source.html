46 void RingBufferDMA::start(void (*call_dma_isr)(void)) {
47 
48  // set up a DMA channel to store the ADC data
49  // The idea is to have ADC_RA as a source,
50  // the buffer as a circular buffer
51  // each ADC conversion triggers a DMA transfer (transferCount(b_size)), of size 2 bytes (transferSize(2))
52 
53  dmaChannel->source(*ADC_RA);
54 
55  dmaChannel->destinationCircular((uint16_t*)p_elems, 2*b_size); // 2*b_size is necessary for some reason
56 
57  dmaChannel->transferSize(2); // both SRC and DST size
58 
59  dmaChannel->transferCount(b_size); // transfer b_size values
60 
61  dmaChannel->interruptAtCompletion(); //interruptAtHalf or interruptAtCompletion
62 
63 
64  uint8_t DMAMUX_SOURCE_ADC = DMAMUX_SOURCE_ADC0;
65  #if ADC_NUM_ADCS>=2
66  if(ADC_number==1){
67  DMAMUX_SOURCE_ADC = DMAMUX_SOURCE_ADC1;
68  }
69  #endif // ADC_NUM_ADCS
70 
71 
72  dmaChannel->triggerAtHardwareEvent(DMAMUX_SOURCE_ADC); // start DMA channel when ADC finishes a conversion
73  dmaChannel->enable();
74 
75  dmaChannel->attachInterrupt(call_dma_isr);
76 
77  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN));
78 }
