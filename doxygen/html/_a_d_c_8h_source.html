61  // Workload-based dispatch policy:
62  // Dispatch conversion to the selected ADC. If no specific ADC is selected (ADC_NUM::ANY):
63  // Check which ADC can handle the pin, if both:
64  // Use the ADC with lesser workload
65  // If only one ADC can measure pin, use it.
66  template<typename ret_type, typename... Args>
67  ret_type workload_dispatch_policy(bool (ADC_Module::*check_fun)(Args... args),
68  ret_type (ADC_Module::*conversion_fun)(Args... args), ADC_NUM adc_num, Args... args) {
69  #if ADC_NUM_ADCS==1
70  return (adc0->*conversion_fun)(args...); // use ADC0
71  #else
72  if(adc_num==ADC_NUM::ANY) { // use no ADC in particular
73  // check which ADC can read the pin
74  bool adc0Pin = (adc0->*check_fun)(args...);
75  bool adc1Pin = (adc1->*check_fun)(args...);
76 
77  if(adc0Pin && adc1Pin) { // Both ADCs
78  if( (adc0->num_measurements) > (adc1->num_measurements)) { // use the ADC with less workload
79  return (adc1->*conversion_fun)(args...);
80  } else {
81  return (adc0->*conversion_fun)(args...);
82  }
83  } else if(adc0Pin) { // ADC0
84  return (adc0->*conversion_fun)(args...);
85  } else if(adc1Pin) { // ADC1
86  return (adc1->*conversion_fun)(args...);
87  } else { // pin not valid in any ADC
88  adc0->fail_flag |= ADC_ERROR_WRONG_PIN;
89  adc1->fail_flag |= ADC_ERROR_WRONG_PIN;
90  return ADC_ERROR_VALUE; // all others are invalid
91  }
92  } else { // Use a specific ADC
93  return (adc[static_cast<uint8_t>(adc_num)]->*conversion_fun)(args...);
94  }
95  #endif // ADC_NUM_ADCS
96  }

98  // Simple dispatch policy:
99  // Dispatch conversion to the selected ADC. If no specific ADC is selected (ADC_NUM::ANY):
100  // Check if ADC0 can handle it, if so, use it.
101  // If not, check ADC1, if so, use it.
102  template<typename ret_type, typename... Args>
103  ret_type simple_dispatch_policy(bool (ADC_Module::*check_fun)(Args... args),
104  ret_type (ADC_Module::*conversion_fun)(Args... args), ADC_NUM adc_num, Args... args) {
105  #if ADC_NUM_ADCS==1
106  return (adc0->*conversion_fun)(args...); // use ADC0
107  #else
108  if(adc_num==ADC_NUM::ANY) { // use no ADC in particular
109  // check which ADC can read the pin
110  bool adc0Pin = (adc0->*check_fun)(args...);
111  if(adc0Pin) { // ADC0
112  return (adc0->*conversion_fun)(args...);
113  }
114 
115  bool adc1Pin = (adc1->*check_fun)(args...);
116  if(adc1Pin) { // ADC1
117  return (adc1->*conversion_fun)(args...);
118  }
119 
120  // Not valid for any ADC
121  adc0->fail_flag |= ADC_ERROR_WRONG_PIN;
122  adc1->fail_flag |= ADC_ERROR_WRONG_PIN;
123  return ADC_ERROR_VALUE; // all others are invalid
124  } else { // Use a specific ADC
125  return (adc[static_cast<uint8_t>(adc_num)]->*conversion_fun)(args...);
126  }
127  #endif // ADC_NUM_ADCS
128  }

130  // Change the return function to the policy that you want: workload_dispatch_policy or simple_dispatch_policy.

133  ret_type (ADC_Module::*conversion_fun)(Args... args),
134  ADC_NUM adc_num, Args... args) {
135  return workload_dispatch_policy(check_fun, conversion_fun, adc_num, args...);
136  //return simple_dispatch_policy(check_fun, conversion_fun, adc_num, args...);
137  }
