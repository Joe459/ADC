223 /*Returns the analog values of both pins, measured at the same time by the two ADC modules.
224 * It waits until the value is read and then returns the result as a struct Sync_result,
225 * use Sync_result.result_adc0 and Sync_result.result_adc1.
226 * If a comparison has been set up and fails, it will return ADC_ERROR_VALUE in both fields of the struct.
227 */
228 ADC::Sync_result ADC::analogSynchronizedRead(uint8_t pin0, uint8_t pin1) {
229 
230  Sync_result res = {ADC_ERROR_VALUE, ADC_ERROR_VALUE};
231 
232  // check pins
233  if ( !adc0->checkPin(pin0) ) {
234  adc0->fail_flag |= ADC_ERROR::WRONG_PIN;
235  return res;
236  }
237  if ( !adc1->checkPin(pin1) ) {
238  adc1->fail_flag |= ADC_ERROR::WRONG_PIN;
239  return res;
240  }
241 
242 
243  // check if we are interrupting a measurement, store setting if so.
244  // vars to save the current state of the ADC in case it's in use
245  ADC_Module::ADC_Config old_adc0_config = {0};
246  uint8_t wasADC0InUse = adc0->isConverting(); // is the ADC running now?
247  if(wasADC0InUse) { // this means we're interrupting a conversion
248  // save the current conversion config, the adc isr will restore the adc
249  __disable_irq();
250  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN) );
251  adc0->saveConfig(&old_adc0_config);
252  __enable_irq();
253  }
254  ADC_Module::ADC_Config old_adc1_config = {0};
255  uint8_t wasADC1InUse = adc1->isConverting(); // is the ADC running now?
256  if(wasADC1InUse) { // this means we're interrupting a conversion
257  // save the current conversion config, the adc isr will restore the adc
258  __disable_irq();
259  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN) );
260  adc1->saveConfig(&old_adc1_config);
261  __enable_irq();
262  }
263 
264  // no continuous mode
265  adc0->singleMode();
266  adc1->singleMode();
267 
268  // start both measurements
269  adc0->startReadFast(pin0);
270  adc1->startReadFast(pin1);
271 
272  // wait for both ADCs to finish
273  while( (adc0->isConverting()) || (adc1->isConverting()) ) { // wait for both to finish
274  yield();
275  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN) );
276  }
277 
278 
279  __disable_irq(); // make sure nothing interrupts this part
280  if ( adc0->isComplete() ) { // conversion succeded
281  res.result_adc0 = adc0->readSingle();
282  } else { // comparison was false
283  adc0->fail_flag |= ADC_ERROR::COMPARISON;
284  }
285  if ( adc1->isComplete() ) { // conversion succeded
286  res.result_adc1 = adc1->readSingle();
287  } else { // comparison was false
288  adc1->fail_flag |= ADC_ERROR::COMPARISON;
289  }
290  __enable_irq();
291 
292 
293  // if we interrupted a conversion, set it again
294  if (wasADC0InUse) {
295  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN) );
296  adc0->loadConfig(&old_adc0_config);
297  }
298  if (wasADC1InUse) {
299  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN) );
300  adc1->loadConfig(&old_adc1_config);
301  }
302 
303  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN) );
304 
305  return res;
306 }

308 /*Returns the diff analog values of both sets of pins, measured at the same time by the two ADC modules.
309 * It waits until the value is read and then returns the result as a struct Sync_result,
310 * use Sync_result.result_adc0 and Sync_result.result_adc1.
311 * If a comparison has been set up and fails, it will return ADC_ERROR_VALUE in both fields of the struct.
312 */
313 ADC::Sync_result ADC::analogSynchronizedReadDifferential(uint8_t pin0P, uint8_t pin0N, uint8_t pin1P, uint8_t pin1N) {
314 
315  Sync_result res = {ADC_ERROR_VALUE, ADC_ERROR_VALUE};;
316 
317  // check pins
318  if(!adc0->checkDifferentialPins(pin0P, pin0N)) {
319  adc0->fail_flag |= ADC_ERROR::WRONG_PIN;
320  return res; // all others are invalid
321  }
322  if(!adc1->checkDifferentialPins(pin1P, pin1N)) {
323  adc1->fail_flag |= ADC_ERROR::WRONG_PIN;
324  return res; // all others are invalid
325  }
326 
327  uint8_t resolution0 = adc0->getResolution();
328  uint8_t resolution1 = adc1->getResolution();
329 
330  // check if we are interrupting a measurement, store setting if so.
331  // vars to save the current state of the ADC in case it's in use
332  ADC_Module::ADC_Config old_adc0_config = {0};
333  uint8_t wasADC0InUse = adc0->isConverting(); // is the ADC running now?
334  if(wasADC0InUse) { // this means we're interrupting a conversion
335  // save the current conversion config, the adc isr will restore the adc
336  __disable_irq();
337  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN) );
338  adc0->saveConfig(&old_adc0_config);
339  __enable_irq();
340  }
341  ADC_Module::ADC_Config old_adc1_config = {0};
342  uint8_t wasADC1InUse = adc1->isConverting(); // is the ADC running now?
343  if(wasADC1InUse) { // this means we're interrupting a conversion
344  // save the current conversion config, the adc isr will restore the adc
345  __disable_irq();
346  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN) );
347  adc1->saveConfig(&old_adc1_config);
348  __enable_irq();
349  }
350 
351  // no continuous mode
352  adc0->singleMode();
353  adc1->singleMode();
354 
355  // start both measurements
356  adc0->startDifferentialFast(pin0P, pin0N);
357  adc1->startDifferentialFast(pin1P, pin1N);
358 
359  // wait for both ADCs to finish
360  while( (adc0->isConverting()) || (adc1->isConverting()) ) {
361  yield();
362  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN) );
363  }
364  __disable_irq(); // make sure nothing interrupts this part
365  if (adc0->isComplete()) { // conversion succeded
366  res.result_adc0 = adc0->readSingle();
367  if(resolution0==16) { // 16 bit differential is actually 15 bit + 1 bit sign
368  res.result_adc0 *= 2; // multiply by 2 as if it were really 16 bits, so that getMaxValue gives a correct value.
369  }
370  } else { // comparison was false
371  adc0->fail_flag |= ADC_ERROR::COMPARISON;
372  }
373  if (adc1->isComplete()) { // conversion succeded
374  res.result_adc1 = adc1->readSingle();
375  if(resolution1==16) { // 16 bit differential is actually 15 bit + 1 bit sign
376  res.result_adc1 *= 2; // multiply by 2 as if it were really 16 bits, so that getMaxValue gives a correct value.
377  }
378  } else { // comparison was false
379  adc1->fail_flag |= ADC_ERROR::COMPARISON;
380  }
381  __enable_irq();
382 
383 
384  // if we interrupted a conversion, set it again
385  if (wasADC0InUse) {
386  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN) );
387  adc0->loadConfig(&old_adc0_config);
388  }
389  if (wasADC1InUse) {
390  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN) );
391  adc1->loadConfig(&old_adc1_config);
392  }
393 
394  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN) );
395 
396  return res;
397 }

402 /* It returns inmediately, get value with readSynchronizedSingle().
403 * If the pin is incorrect it returns false
404 * If this function interrupts a measurement, it stores the settings in adc_config
405 */
406 bool ADC::startSynchronizedSingleRead(uint8_t pin0, uint8_t pin1) {
407 
408  // check pins
409  if ( !adc0->checkPin(pin0) ) {
410  adc0->fail_flag |= ADC_ERROR::WRONG_PIN;
411  return false;
412  }
413  if ( !adc1->checkPin(pin1) ) {
414  adc1->fail_flag |= ADC_ERROR::WRONG_PIN;
415  return false;
416  }
417 
418  // check if we are interrupting a measurement, store setting if so.
419  adc0->adcWasInUse = adc0->isConverting(); // is the ADC running now?
420  if(adc0->adcWasInUse) { // this means we're interrupting a conversion
421  // save the current conversion config, the adc isr will restore the adc
422  __disable_irq();
423  //digitalWriteFast(LED_BUILTIN, !digitalReadFast(LED_BUILTIN) );
424  adc0->saveConfig(&adc0->adc_config);
425  __enable_irq();
426  }
427  adc1->adcWasInUse = adc1->isConverting(); // is the ADC running now?
428  if(adc1->adcWasInUse) { // this means we're interrupting a conversion
